Range Slider

Default

Custom range slider:

Usage

    <div style="width: 250px">
<cdg-range-slider min="0" max="1000" value="50"></cdg-range-slider>
</div>

const rangeSlider = document.querySelector('cdg-range-slider');
rangeSlider.addEventListener('change', (event) => {
    const newValue = event.detail.value;
    console.log('Slider value changed:', newValue);
  });
