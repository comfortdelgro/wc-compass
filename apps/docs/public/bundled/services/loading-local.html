By clicking Show local button, the loading indicator will be shown on .page-container element and will be disapeared in the next 3 seconds 

      <button class="cdg-button" id="local-button">Show Local</button>
<script>
    function showLocal() {
        const host = document.querySelector('.page-container')
        const loadingId = cdgLoadingService.show('local', host)
        setTimeout(() => {
            cdgLoadingService.hide(loadingId)
        }, 3000)
    }

    const button = this.querySelector('#local-button')
    button.addEventListener('click', this.showLocal.bind(this))
</script>
