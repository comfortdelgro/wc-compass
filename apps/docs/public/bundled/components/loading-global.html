By clicking Show Global button, the loading indicator will be shown on body element, it will covers everything and will be disapeared in the next 3 seconds 

      <button class="cdg-button" id="global-button">Show Global</button>
<script>
  function showLocal() {
    const loadingId = cdgLoadingService.show('global')
    setTimeout(() => {
      cdgLoadingService.hide(loadingId)
    }, 3000)
  }

  const button = this.querySelector('#global-button')
  button.addEventListener('click', showLocal)
</script>
