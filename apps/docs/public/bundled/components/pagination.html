Pagination

Default

  <cdg-pagination
  id="simple-pagination"
  current-page="1"
  total="30"
  page-size="10"
></cdg-pagination>
<script>
const simplePaging = document.querySelector('#simple-pagination');

simplePaging.addEventListener('navigate', (data) => {
  simplePaging.currentPage = data.detail;
});
</script>
