    <script>
      const contextMenus = document.querySelectorAll('cdg-context-menu')
      for (let index = 0; index < contextMenus.length; index++) {
        const contextMenu = contextMenus.item(index)
        contextMenu.addEventListener('onContextMenuOpen', () => {
          console.log(`Context menu "${contextMenu.getAttribute('name')}" opened`)
        })
        contextMenu.addEventListener('onContextMenuClose', () => {
          console.log(`Context menu "${contextMenu.getAttribute('name')}" closed`)
        })
      }
    </script>
