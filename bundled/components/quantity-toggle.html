The quantity toggle component is used to increase or decrease an incremental venue, usually best used when the user needs to enter or adjust the quantity of a selected item. 

    <cdg-quantity-toggle onchange="handleOnchange(event)">
    <button class="cdg-button">-</button>
    <input type="number" placeholder="Input quantity" value="0" />
    <button class="cdg-button">+</button>
</cdg-quantity-toggle>

<script>
function handleOnchange (event) {
    cdgToastService.toast('Changed ' + event.target.value);
};
</script>

    <style>
/** To customize size. It should not be smaller than 160px **/
.sample-size {
    width: 160px;
}
</style>
<cdg-quantity-toggle max="10" min="1" id="maxMinSample" class="sample-size">
    <button class="cdg-button">-</button>
    <input type="number" placeholder="Input quantity" value="0" />
    <button class="cdg-button">+</button>
</cdg-quantity-toggle>
<script>
  const quantityToggle = document.querySelector('#maxMinSample');
  quantityToggle.addEventListener('validChange', (event) => {
    cdgToastService.toast('Error: ' + JSON.stringify(event.detail));
  });
</script>
