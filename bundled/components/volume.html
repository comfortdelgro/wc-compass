<script>
    const volumeControl = document.querySelector('cdg-volume');
    volumeControl.addEventListener('volumechange', (event) => {
        cdgToastService.toast('Volume: ' + event.detail);
    });
    volumeControl.addEventListener('mute', (event) => {
        cdgToastService.toast('Mute: ' + event.detail);
    });
</script>
