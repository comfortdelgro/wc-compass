    <div class="cdg-right-group">
        <button class="cdg-button" id="zoomIn">Zoom In</button>
        <button class="cdg-button" id="zoomOut">Zoom Out</button>
    </div>

    id="inlineImageViewer"
    src="https://picsum.photos/id/211/2000/1400"
    style="aspect-ratio: 1.7"
></cdg-image-viewer>

<script>
const imageViewer = document.querySelector('#inlineImageViewer')
const zoomInButton = document.querySelector('#zoomIn')
const zoomOutButton = document.querySelector('#zoomOut')

zoomInButton.addEventListener('click', () => {
    imageViewer.zoomIn()
})

zoomOutButton.addEventListener('click', () => {
    imageViewer.zoomOut()
})
</script>
